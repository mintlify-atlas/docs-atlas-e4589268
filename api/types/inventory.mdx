## Asset

Represents a physical asset (equipment) in the system with complete maintenance history and metadata.

```typescript
interface Asset {
  id: string;
  name: string;
  tag: string;
  type: string;
  equipment_class: string;
  manufacturer: string;
  model_number: string;
  serial_number: string;
  criticality_level: string;
  operating_hours: string;
  commissioned_date: string;
  status: string;
  maintenance_strategy: string;
  last_performed_maintenance: string;
  major_overhaul: string;
  last_date_overhaul: string;
  power_rating: string;
  speed: string;
  capacity: string;
  is_modified: boolean;
  datasheet: {
    file_url: string;
    file_name: string;
    uploaded_at: string;
  } | null;
  assignee: AssigneeDetails;
  parent_site: SiteReference;
  created_at: number;
  created_at_datetime: string;
  updated_at: number;
  updated_at_datetime: string;
}
```

## Sites

Represents a physical site or facility that contains assets.

```typescript
interface Sites {
  id: string;
  name: string;
  tag: string;
  description: string;
  address: string;
  city: string;
  country: string;
  installation_environment: string;
  regulations_and_standards: string[];
  manager_name: string;
  manager_email: string;
  manager_phone_number: string;
  manager_location: string;
  organization: OrganizationReference;
  attachments: unknown;
  members: unknown;
  created_at: number;
  created_at_datetime: string;
  updated_at: number;
  updated_at_datetime: string;
}
```

## SitesAnalytics

Analytics data for sites with issues tracking.

```typescript
interface SitesAnalytics {
  total_sites: number;
  sites_trend_percentage: number;
  sites_with_issues: number;
  issues_trend_percentage: number;
}
```

### Fields

<ResponseField name="total_sites" type="number" required>
  Total number of sites in the organization
</ResponseField>

<ResponseField name="sites_trend_percentage" type="number" required>
  Percentage change in total sites compared to previous period
</ResponseField>

<ResponseField name="sites_with_issues" type="number" required>
  Number of sites with active issues or alarms
</ResponseField>

<ResponseField name="issues_trend_percentage" type="number" required>
  Percentage change in sites with issues
</ResponseField>

## AssetAnalytics

Comprehensive analytics for asset inventory and health monitoring.

```typescript
interface AssetAnalytics {
  total_assets: number;
  asset_trend: {
    growth: number;
    percentage: number;
  };
  critical_assets: {
    total: number;
    trend: {
      growth: number;
      percentage: number;
    };
  };
  assets_with_alarms: {
    total: number;
    trend: {
      growth: number;
      percentage: number;
    };
  };
}
```

### Fields

<ResponseField name="total_assets" type="number" required>
  Total number of assets
</ResponseField>

<ResponseField name="asset_trend" type="object" required>
  Overall asset count trend
  
  <Expandable title="Trend object">
    <ResponseField name="growth" type="number">
      Absolute change in asset count
    </ResponseField>
    <ResponseField name="percentage" type="number">
      Percentage change
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="critical_assets" type="object" required>
  Critical asset metrics
  
  <Expandable title="Critical assets">
    <ResponseField name="total" type="number">
      Number of critical assets
    </ResponseField>
    <ResponseField name="trend" type="object">
      Trend data with growth and percentage fields
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="assets_with_alarms" type="object" required>
  Assets with active alarms
  
  <Expandable title="Assets with alarms">
    <ResponseField name="total" type="number">
      Number of assets with alarms
    </ResponseField>
    <ResponseField name="trend" type="object">
      Trend data with growth and percentage fields
    </ResponseField>
  </Expandable>
</ResponseField>

## Usage Example

```typescript
import type { Asset, Sites, AssetAnalytics } from '@/types';

// Fetch and type asset data
const asset: Asset = await getAssetById(assetId);

console.log(`Asset: ${asset.name} (${asset.tag})`);
console.log(`Criticality: ${asset.criticality_level}`);
console.log(`Site: ${asset.parent_site.name}`);

// Use analytics
const analytics: AssetAnalytics = await getAssetAnalytics();
console.log(`Total: ${analytics.total_assets}`);
console.log(`Critical: ${analytics.critical_assets.total}`);
```
